let
  // define function with input parameters
  fn_GetReportUsers = (objectType as text, objectId as text, accessToken as text) =>
    let
    // define source as the base REST API URL
	Source = Json.Document(Web.Contents("https://api.powerbi.com/v1.0/myorg/",
    [
    // add relative path for the object type
	RelativePath = "admin/" & objectType & "/" & objectId & "/users",
    // define the headers to be included in the web call
	Headers=[Authorization="Bearer " & accessToken]
    ] )),
    value = Source[value]
in
    value
in
  fn_GetReportUsers
